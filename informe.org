    BackgroundColor lightgreen

    BackgroundColor lightyellow
    BorderColor black
}

title System Architecture Overview

package "Client Layer" {
    component [Mobile Application\n(React Native)] as mobile
    component [Backoffice WebApp\n(React)] as webapp
}

package "Backend Services" {
    component [Main Backend Service\n(Node.js)\n\n- API Gateway\n- Business Logic\n- Authentication\n- Data Management] as mainapi
    component [ML Service\n(Python)\n\n- ML Algorithms\n- Data Processing\n- Model Training\n- Predictions] as mlservice
}

package "Data Layer" {
    database [Database\n(PostgreSQL/MongoDB)] as db
    database [ML Models\n& Training Data] as mldata
}

cloud "External Services" as external {
    component [Third-party APIs] as thirdparty
    component [Cloud Storage] as storage
}

' Relationships
mobile --> mainapi : REST/GraphQL API
webapp --> mainapi : REST/GraphQL API
mainapi --> db : Database Operations
mainapi --> mlservice : ML Requests
mlservice --> mldata : Model & Data Access
mainapi --> thirdparty : External Integrations
mainapi --> storage : File Storage
mlservice --> storage : Model Storage

' Notes
note right of mobile
  - iOS & Android support
  - Real-time updates
  - Offline capabilities
  - Push notifications
end note

note right of webapp
  - Admin dashboard
  - Analytics & reporting
  - User management
  - System configuration
end note

note bottom of mainapi
  - Authentication & authorization
  - Rate limiting
  - Logging & monitoring
  - Load balancing ready
end note

note bottom of mlservice
  - Model serving
  - Batch processing
  - Real-time predictions
  - A/B testing support
end note
